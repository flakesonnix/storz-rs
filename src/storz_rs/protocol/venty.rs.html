19/// Venty or Veazy device, both share the same protocol.
20pub struct Venty {
21    peripheral: Peripheral,
22    model: DeviceModel,
23    state: Arc<Mutex<DeviceState>>,
24    state_tx: broadcast::Sender<DeviceState>,
25}
26
27impl Venty {
28    pub async fn new(peripheral: Peripheral, model: DeviceModel) -> Result<Self, StorzError> {
29        let (state_tx, _) = broadcast::channel(16);
30
31        let device = Self {
32            peripheral,
33            model,
34            state: Arc::new(Mutex::new(DeviceState::default())),
35            state_tx,
36        };
37
38        device.init().await?;
39        device.spawn_notification_loop();
40        Ok(device)
41    }
42
43    async fn control_characteristic(&self) -> Result<btleplug::api::Characteristic, StorzError> {
44        self.peripheral
45            .characteristics()
46            .into_iter()
47            .find(|c| c.uuid == VENTY_CONTROL)
48            .ok_or_else(|| StorzError::ParseError("Venty control characteristic not found".into()))
49    }
50
51    async fn write_command(&self, buf: &[u8]) -> Result<(), StorzError> {
52        let ch = self.control_characteristic().await?;
53        self.peripheral
54            .write(&ch, buf, WriteType::WithoutResponse)
55            .await?;
56        Ok(())
57    }
58
59    async fn init(&self) -> Result<(), StorzError> {
60        let ch = self.control_characteristic().await?;
61
62        // Subscribe to notifications first
63        self.peripheral.subscribe(&ch).await?;
64        debug!("Subscribed to Venty/Veazy control notifications");
65
66        // Send initialization sequence: 0x02, 0x1D, 0x01, 0x04
67        for &cmd in &[0x02u8, 0x1Du8, 0x01u8, 0x04u8] {
68            let buf = utils::build_venty_command(cmd, 0, &[]);
69            self.write_command(&buf).await?;
70            debug!("Venty/Veazy init command 0x{cmd:02X} sent");
71        }
72
73        Ok(())
74    }
75
76    fn spawn_notification_loop(&self) {
77        let peripheral = self.peripheral.clone();
78        let state = self.state.clone();
79        let state_tx = self.state_tx.clone();
80        let model = self.model;
81
82        tokio::spawn(async move {
83            let mut stream = match peripheral.notifications().await {
84                Ok(s) => s,
85                Err(e) => {
86                    warn!("Venty/Veazy failed to start notification stream: {e}");
87                    return;
88                }
89            };
90
91            while let Some(data) = stream.next().await {
92                Self::debug_dump_notification(&data.value);
93                Self::handle_notification_inner(&state, &state_tx, &data.value);
94            }
95
96            warn!("{model} notification stream ended (disconnect?)");
97        });
98    }
99
100    fn debug_dump_notification(data: &[u8]) {
101        if data.is_empty() {
102            debug!("Venty notification: empty");
103            return;
104        }
105        let cmd = data[0];
106        let hex: String = data.iter().map(|b| format!("{b:02X}")).collect::<Vec<_>>().join(" ");
107        debug!("Venty notification: cmd=0x{cmd:02X} len={} [{hex}]", data.len());
108
109        if data.len() >= 2 {
110            debug!("  [0] cmd_id    = 0x{:02X}", data[0]);
111            debug!("  [1] mask      = 0x{:02X} ({:08b})", data[1], data[1]);
112        }
113        if data.len() >= 4 {
114            let raw23 = u16::from_le_bytes([data[2], data[3]]);
115            debug!("  [2:3] bytes   = 0x{:04X} (u16={raw23}, as_temp={:.1}°C — UNUSED)", raw23, raw23 as f32 / 10.0);
116        }
117        if data.len() >= 6 {
118            let raw45 = u16::from_le_bytes([data[4], data[5]]);
119            debug!("  [4:5] target  = 0x{:04X} (u16={raw45}, {:.1}°C)", raw45, raw45 as f32 / 10.0);
120        }
121        if data.len() >= 7 {
122            debug!("  [6]   boost   = {}°C", data[6]);
123        }
124        if data.len() >= 8 {
125            debug!("  [7]   sboost  = {}°C", data[7]);
126        }
127        if data.len() >= 9 {
128            debug!("  [8]   battery = {}%", data[8]);
129        }
130        if data.len() >= 11 {
131            let timer = data[9] as u16 + (data[10] as u16) * 256;
132            debug!("  [9:10] timer  = {timer}s");
133        }
134        if data.len() >= 12 {
135            let mode = match data[11] {
136                0 => "off",
137                1 => "normal",
138                2 => "boost",
139                3 => "superboost",
140                _ => "unknown",
141            };
142            debug!("  [11]  heater  = {mode}");
143        }
144        if data.len() >= 14 {
145            debug!("  [13]  charger = {}", data[13]);
146        }
147        if data.len() >= 15 {
148            let s = data[14];
149            debug!("  [14]  settings= 0x{s:02X} (cel={},setpoint={},vibrate={})", 
150                if s & 1 == 0 { "yes" } else { "no" },
151                if s & 2 != 0 { "yes" } else { "no" },
152                if s & 0x40 != 0 { "yes" } else { "no" });
153        }
154    }
155
156    fn handle_notification_inner(
157        state: &Arc<Mutex<DeviceState>>,
158        state_tx: &broadcast::Sender<DeviceState>,
159        data: &[u8],
160    ) {
161        if data.is_empty() {
162            return;
163        }
164
165        let mut state = match state.try_lock() {
166            Ok(s) => s,
167            Err(_) => return,
168        };
169
170        let cmd_id = data[0];
171
172        match cmd_id {
173            0x01 | 0x05 if data.len() >= 15 => {
174                // Bytes 4+5: target temperature (u16 LE, /10)
175                if data.len() >= 6 {
176                    let raw = u16::from_le_bytes([data[4], data[5]]);
177                    state.target_temp = Some((raw as f32) / 10.0);
178                }
179
180                // Byte 11: heater mode (0=off, 1=normal, 2=boost, 3=superboost)
181                if data.len() >= 12 {
182                    state.heater_on = data[11] > 0;
183                }
184
185                // Byte 14, bit 1: setpoint reached
186                if data.len() >= 15 {
187                    state.setpoint_reached = (data[14] & 0x02) != 0;
188                }
189
190                // Venty/Veazy don't have pumps
191                state.pump_on = false;
192
193                let _ = state_tx.send(state.clone());
194            }
195            _ => {
196                debug!(
197                    "Venty/Veazy unhandled notification cmd=0x{:02X} len={}",
198                    cmd_id,
199                    data.len()
200                );
201            }
202        }
203    }
204}

206#[async_trait]
207impl VaporizerControl for Venty {
208    async fn get_current_temperature(&self) -> Result<f32, StorzError> {
209        // Venty doesn't expose current temp directly via a read;
210        // it comes through notifications. Return cached value.
211        let state = self.state.lock().await;
212        state.current_temp.ok_or(StorzError::ParseError(
213            "Current temperature not yet available from device notifications".into(),
214        ))
215    }
216
217    async fn get_target_temperature(&self) -> Result<f32, StorzError> {
218        let state = self.state.lock().await;
219        state.target_temp.ok_or(StorzError::ParseError(
220            "Target temperature not yet available from device notifications".into(),
221        ))
222    }
223
224    async fn set_target_temperature(&self, celsius: f32) -> Result<(), StorzError> {
225        let raw = (celsius * 10.0).round() as u16;
226        let low = (raw & 0xFF) as u8;
227        let high = ((raw >> 8) & 0xFF) as u8;
228
229        let buf = utils::build_venty_command(
230            0x01, // Write command
231            0x02, // SET_TEMPERATURE mask
232            &[(4, low), (5, high)],
233        );
234        self.write_command(&buf).await?;
235        debug!("Venty/Veazy target temp set to {celsius}°C");
236        Ok(())
237    }
238
239    async fn heater_on(&self) -> Result<(), StorzError> {
240        let buf = utils::build_venty_command(
241            0x01,
242            0x20,       // HEATER mask
243            &[(11, 1)], // heater mode = 1 (normal)
244        );
245        self.write_command(&buf).await?;
246        debug!("Venty/Veazy heater ON");
247        Ok(())
248    }
249
250    async fn heater_off(&self) -> Result<(), StorzError> {
251        let buf = utils::build_venty_command(
252            0x01,
253            0x20,       // HEATER mask
254            &[(11, 0)], // heater mode = 0 (off)
255        );
256        self.write_command(&buf).await?;
257        debug!("Venty/Veazy heater OFF");
258        Ok(())
259    }
260
261    async fn pump_on(&self) -> Result<(), StorzError> {
262        Err(StorzError::UnsupportedOperation {
263            device: self.model.to_string(),
264            operation: "pump_on".into(),
265        })
266    }
267
268    async fn pump_off(&self) -> Result<(), StorzError> {
269        Err(StorzError::UnsupportedOperation {
270            device: self.model.to_string(),
271            operation: "pump_off".into(),
272        })
273    }
274
275    async fn get_state(&self) -> Result<DeviceState, StorzError> {
276        let state = self.state.lock().await;
277        Ok(state.clone())
278    }
279
280    async fn subscribe_state(
281        &self,
282    ) -> Result<Pin<Box<dyn Stream<Item = DeviceState> + Send>>, StorzError> {
283        let rx = self.state_tx.subscribe();
284        Ok(Box::pin(
285            BroadcastStream::new(rx).filter_map(|r| async move { r.ok() }),
286        ))
287    }
288
289    fn device_model(&self) -> DeviceModel {
290        self.model
291    }
292}
